4. Controlled Access via Functions
Expose tenant context only through controlled functions:

CREATE OR REPLACE FUNCTION evik_platform.get_user_tenant(p_user_id uuid)
RETURNS uuid
SECURITY DEFINER
AS $$
    SELECT tenant_id
    FROM evik_platform.user_tenant_map
    WHERE user_id = p_user_id;
$$ LANGUAGE sql;


Tenant roles can be granted EXECUTE on this function, but not direct access to the table:

GRANT EXECUTE ON FUNCTION evik_platform.get_user_tenant(uuid) TO tenant_user;

CREATE TABLE evik_platform.command_audit (
    audit_id     UUID PRIMARY KEY DEFAULT gen_random_uuid(), -- unique identifier
    tenant_id    UUID NOT NULL,                             -- tenant context
    user_id      UUID NOT NULL,                             -- user executing command
    command_text TEXT NOT NULL,                             -- raw SQL or API command
    command_type TEXT NOT NULL,                             -- e.g., SELECT, INSERT, UPDATE, DELETE
    version      BIGINT NOT NULL,                           -- sequential version number
    executed_at  TIMESTAMPTZ NOT NULL DEFAULT now(),        -- timestamp of execution
    success      BOOLEAN NOT NULL                           -- true if succeeded, false if failed
);

GRANT SELECT ON evik_platform.command_audit TO superadmin;
GRANT SELECT, INSERT ON evik_platform.command_audit TO tenant_owner;

CREATE OR REPLACE FUNCTION evik_platform.command_audit_trigger()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    INSERT INTO evik_platform.command_audit (
        tenant_id,
        user_id,
        command_text,
        command_type,
        version,
        success
    )
    VALUES (
        -- tenant context from user_tenant_map
        (SELECT tenant_id FROM evik_platform.user_tenant_map WHERE db_role = current_user),
        -- user context (assuming your user_account table has id for current_user)
        (SELECT id FROM evik_platform.user_account WHERE db_role = current_user),
        current_query(),   -- full SQL text executed
        TG_OP,             -- command type: INSERT, UPDATE, DELETE
        1,                 -- versioning, you can increment or set default
        TRUE               -- mark success, can be set dynamically
    );
    RETURN NEW;
END;
$$;

ON evik_platform.events

        sensitivity = 'public'

3. Trigger for Versioning

CREATE OR REPLACE FUNCTION evik_platform.versioning_trigger()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    -- On update, archive old row into history
    IF TG_OP = 'UPDATE' THEN
        INSERT INTO evik_platform.industry_lifecycle_history (
            lifecycle_id, industry_id, sector_id, stage,
            version, lifecycle_status, effective_from, effective_to
        )
        VALUES (
            OLD.id, OLD.industry_id, OLD.sector_id, OLD.stage,
            OLD.version, OLD.lifecycle_status, OLD.effective_from, OLD.effective_to
        );

        -- Increment version in main table
        NEW.version := OLD.version + 1;
    END IF;

    RETURN NEW;
END;
$$;

CREATE TRIGGER industry_lifecycle_versioning
BEFORE UPDATE ON evik_platform.industry_lifecycle
FOR EACH ROW
EXECUTE FUNCTION evik_platform.versioning_trigger();

CREATE OR REPLACE FUNCTION evik_platform.tenants_audit()
RETURNS TRIGGER AS $$
DECLARE
    next_version INT;
BEGIN
    -- Determine next version number
    SELECT COALESCE(MAX(version), 0) + 1
    INTO next_version
    FROM evik_platform.tenants_history
    WHERE tenant_id = OLD.id;

    -- Insert old record into history table
    INSERT INTO evik_platform.tenants_history (tenant_id, version, data, changed_by, action)
    VALUES (
        OLD.id,
        next_version,
        to_jsonb(OLD),
        current_user,
        TG_OP
    );

    -- Update version in main table
    NEW.version := next_version;

    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER tenants_audit_trigger
AFTER UPDATE OR DELETE ON evik_platform.tenants
FOR EACH ROW
EXECUTE FUNCTION evik_platform.tenants_audit();

CREATE OR REPLACE FUNCTION evik_platform.command_audit_trigger()
RETURNS trigger
LANGUAGE plpgsql
AS $$
BEGIN
    INSERT INTO evik_platform.command_audit (
        tenant_id,
        user_id,
        command_text,
        command_type,
        version,
        executed_at,
        success
    )
    VALUES (
        -- tenant context
        (SELECT tenant_id FROM evik_platform.user_tenant_map WHERE db_role = current_user),
        -- user context
        (SELECT id FROM evik_platform.user_account WHERE db_role = current_user),
        current_query(),   -- full SQL text executed
        TG_OP,             -- INSERT / UPDATE / DELETE
        COALESCE(NEW.version, 1), -- versioning if present
        now(),
        TRUE
    );
    RETURN NEW;
END;
$$;

CREATE OR REPLACE FUNCTION evik_platform.audit_user_sessions()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    INSERT INTO evik_audit.audit_ledger (
        tenant_id,
        timestamp,
        actor_id,
        actor_role,
        command_name,
        resource_type,
        resource_id,
        summary,
        severity,
        metadata
    )
    VALUES (
        COALESCE(NEW.tenant_id, OLD.tenant_id),
        now(),
        -- actor_id: map current_user to user_id if available
        COALESCE(NEW.user_id, OLD.user_id),
        current_user,
        TG_OP,  -- command_name: INSERT / UPDATE / DELETE
        'user_session',
        COALESCE(NEW.session_id, OLD.session_id),
        CASE
            WHEN TG_OP = 'INSERT' THEN 'Session created'
            WHEN TG_OP = 'UPDATE' AND NEW.revoked = true THEN 'Session revoked'
            WHEN TG_OP = 'UPDATE' THEN 'Session updated'
            WHEN TG_OP = 'DELETE' THEN 'Session deleted'
        END,
        'info',  -- severity: default to info, can be escalated if needed
        CASE
            WHEN TG_OP = 'INSERT' THEN row_to_json(NEW)::jsonb
            WHEN TG_OP = 'UPDATE' THEN jsonb_build_object(
                'old', row_to_json(OLD)::jsonb,
                'new', row_to_json(NEW)::jsonb
            )
            WHEN TG_OP = 'DELETE' THEN row_to_json(OLD)::jsonb
        END
    );
    RETURN NULL;
END;
$$;
