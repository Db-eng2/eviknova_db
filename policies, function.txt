CREATE OR REPLACE FUNCTION evik_audit.prevent_modifications()
RETURNS trigger AS $$
BEGIN
    RAISE EXCEPTION 'Audit table is append-only. Updates/Deletes are not allowed.';
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER audit_no_update_delete
BEFORE UPDATE OR DELETE ON evik_audit.audit_ledger
FOR EACH ROW EXECUTE FUNCTION evik_audit.prevent_modifications();

3) Corrected Write Path.

All writes go through SECURITY DEFINER functions that take tenant_id explicitly:

CREATE OR REPLACE FUNCTION evik_platform.secure_insert_user_session(
    p_tenant_id uuid,
    p_user_id uuid,
    p_metadata jsonb
)
RETURNS void
SECURITY DEFINER
AS $$
BEGIN
    INSERT INTO evik_platform.user_sessions(tenant_id, user_id, metadata)
    VALUES (p_tenant_id, p_user_id, p_metadata);

    INSERT INTO evik_audit.audit_ledger(actor_id, tenant_id, action, created_at)
    VALUES (current_user, p_tenant_id, 'INSERT user_session', now());
END;
$$ LANGUAGE plpgsql;

Correct Grant Script.

-- Revoke any broad access
REVOKE ALL ON FUNCTION evik_platform.secure_insert_user_session(uuid, uuid, jsonb) FROM PUBLIC;

-- Allow only the service writer role to execute
GRANT EXECUTE ON FUNCTION evik_platform.secure_insert_user_session(uuid, uuid, jsonb)
TO evik_service_writer;

CREATE OR REPLACE FUNCTION evik_platform.get_actor_role()
RETURNS text
SECURITY DEFINER
AS $$
DECLARE
    v_role text;
BEGIN
    SELECT db_role
    INTO v_role
    FROM evik_platform.user_tenant_map
    WHERE actor_id = evik_platform.get_actor_context()
    LIMIT 1;

    RETURN v_role;
END;
$$ LANGUAGE plpgsql;


select 

CREATE POLICY abac_event_access_select
ON evik_platform.events
FOR SELECT
TO PUBLIC
USING (
    tenant_id = evik_platform.get_tenant_context()
    AND (
        sensitivity = 'public'
        OR (sensitivity = 'internal' AND evik_platform.get_actor_context() IS NOT NULL)
        OR (sensitivity = 'restricted' AND evik_platform.get_actor_role() = 'compliance_officer')
    )
);

REVOKE EXECUTE ON FUNCTION evik_platform.get_actor_role() FROM PUBLIC;
GRANT EXECUTE ON FUNCTION evik_platform.get_actor_role() TO evik_service_writer;


ABAC Policy for INSERT 

CREATE POLICY abac_event_access_insert
ON evik_platform.events
FOR INSERT
TO PUBLIC
WITH CHECK (
    tenant_id = evik_platform.get_tenant_context()
    AND (
        -- Only allow restricted events to be created by compliance_officer
        (sensitivity = 'restricted' AND evik_platform.get_actor_role() = 'compliance_officer')
        OR (sensitivity IN ('public','internal'))
    )
);


UPDATE

CREATE POLICY abac_event_access_update
ON evik_platform.events
FOR UPDATE
TO PUBLIC
USING (
    tenant_id = evik_platform.get_tenant_context()
)
WITH CHECK (
    tenant_id = evik_platform.get_tenant_context()
    AND (
        -- Restricted events can only be updated by compliance_officer
        (sensitivity = 'restricted' AND evik_platform.get_actor_role() = 'compliance_officer')
        OR (sensitivity IN ('public','internal'))
    )
);

CREATE POLICY abac_resource_access
ON evik_platform.resources
FOR SELECT
USING (
    tenant_id = evik_platform.get_tenant_context()
    AND EXISTS (
        SELECT 1
        FROM evik_platform.user_metadata um
        JOIN evik_platform.resource_metadata rm
          ON rm.tenant_id = um.tenant_id
        WHERE um.user_id = evik_platform.get_actor_context()
          AND rm.resource_id = evik_platform.resources.id
          AND um.country = rm.country
          AND um.clearance IN ('high','medium')   -- clearance check
          AND rm.sensitivity != 'restricted'      -- block restricted unless special role
    )
);

1) DROP FUNCTION IF EXISTS evik_platform.set_tenant_context(uuid);

-- 2. Create secure binding function (still called set_tenant_context)
-- This version validates actor ↔ tenant mapping before setting context

CREATE OR REPLACE FUNCTION evik_platform.set_tenant_context(p_actor_id uuid, p_tenant_id uuid)
RETURNS void
AS $$
BEGIN
    -- Validate actor ↔ tenant relationship
    IF NOT EXISTS (
        SELECT 1
        FROM evik_platform.user_tenant_map
        WHERE user_id = p_actor_id
          AND tenant_id = p_tenant_id
    ) THEN
        RAISE EXCEPTION 'Actor % is not authorized for tenant %', p_actor_id, p_tenant_id;
    END IF;

    -- Set tenant context only after validation (transaction-scoped)
    EXECUTE format('SET LOCAL evik.tenant_context = %L', p_tenant_id::text);
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

REVOKE EXECUTE ON FUNCTION evik_platform.set_tenant_context(uuid, uuid) FROM PUBLIC;
GRANT EXECUTE ON FUNCTION evik_platform.set_tenant_context(uuid, uuid) TO evik_service_writer;


-- 3. Refined getter with enforcement

CREATE OR REPLACE FUNCTION evik_platform.get_tenant_context()
RETURNS uuid
AS $$
DECLARE
    tenant_id uuid;
BEGIN
    -- Try to read tenant context
    BEGIN
        tenant_id := current_setting('evik.tenant_context', true)::uuid;
    EXCEPTION WHEN others THEN
        tenant_id := NULL;
    END;

    -- Enforce: tenant context must be set
    IF tenant_id IS NULL THEN
        RAISE EXCEPTION 'Tenant context not set';
    END IF;

    RETURN tenant_id;
END;
$$ LANGUAGE plpgsql;

REVOKE EXECUTE ON FUNCTION evik_platform.get_tenant_context() FROM PUBLIC;
GRANT EXECUTE ON FUNCTION evik_platform.get_tenant_context() TO evik_service_writer;
